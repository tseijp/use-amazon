# Available book apis
import { useBook, UseBook } from "use-book/src"

```jsx
import { useBook, UseBook } from "use-book/src"
render (
  <UseBook size="xs" as="img" isbn="4041013380" />
  <UseBook size="sm" as="img" isbn="4041013380" isssl />
  <UseBook size="md" as="img" isbn="4041013381" search={10} />
  <UseBook size="lg" isbn="4041013380">
    {({src, alt}) =>
      <img src={src} alt={alt}/>
    }
  </UseBook>
)
```
